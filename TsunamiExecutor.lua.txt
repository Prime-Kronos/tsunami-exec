-- Big toggle row (mobile-friendly 48px height)
local function makeToggle(parent, labelText, desc, order, onToggle)
    local row = Instance.new("Frame")
    row.Size             = UDim2.new(1,0,0,50)
    row.BackgroundColor3 = C.PANEL2
    row.BorderSizePixel  = 0
    row.LayoutOrder      = order
    row.Parent           = parent
    Instance.new("UICorner", row).CornerRadius = UDim.new(0,7)

    local nameL = Instance.new("TextLabel", row)
    nameL.Size  = UDim2.new(1,-70,0,24)
    nameL.Position = UDim2.new(0,12,0,6)
    nameL.BackgroundTransparency = 1
    nameL.Font  = Enum.Font.GothamBold
    nameL.Text  = labelText
    nameL.TextColor3 = C.TEXT
    nameL.TextSize = 13
    nameL.TextXAlignment = Enum.TextXAlignment.Left

    if desc then
        local descL = Instance.new("TextLabel", row)
        descL.Size  = UDim2.new(1,-70,0,16)
        descL.Position = UDim2.new(0,12,0,28)
        descL.BackgroundTransparency = 1
        descL.Font  = Enum.Font.Gotham
        descL.Text  = desc
        descL.TextColor3 = C.SUBTEXT
        descL.TextSize = 10
        descL.TextXAlignment = Enum.TextXAlignment.Left
    end

    -- Toggle switch
    local togBg = Instance.new("Frame", row)
    togBg.Size     = UDim2.new(0,52,0,28)
    togBg.Position = UDim2.new(1,-62,0.5,-14)
    togBg.BackgroundColor3 = C.OFF
    togBg.BorderSizePixel = 0
    Instance.new("UICorner", togBg).CornerRadius = UDim.new(1,0)

    local togKnob = Instance.new("Frame", togBg)
    togKnob.Size     = UDim2.new(0,22,0,22)
    togKnob.Position = UDim2.new(0,3,0,3)
    togKnob.BackgroundColor3 = Color3.fromRGB(120,140,160)
    togKnob.BorderSizePixel = 0
    Instance.new("UICorner", togKnob).CornerRadius = UDim.new(1,0)

    local isOn = false
    local function setOn(v)
        isOn = v
        tween(togBg, {BackgroundColor3 = v and C.NEON or C.OFF}, 0.18)
        tween(togKnob, {
            Position = v and UDim2.new(0,27,0,3) or UDim2.new(0,3,0,3),
            BackgroundColor3 = v and C.WHITE or Color3.fromRGB(120,140,160)
        }, 0.18)
        if v then
            tween(togBg, {BackgroundColor3 = C.NEON}, 0.18)
        end
    end

    local clickBtn = Instance.new("TextButton", row)
    clickBtn.Size = UDim2.new(1,0,1,0)
    clickBtn.BackgroundTransparency = 1
    clickBtn.Text = ""
    clickBtn.MouseButton1Click:Connect(function()
        isOn = not isOn
        setOn(isOn)
        if onToggle then onToggle(isOn) end
        -- ripple
        tween(row, {BackgroundColor3 = Color3.fromRGB(15,22,40)}, 0.1)
        task.delay(0.1, function() tween(row, {BackgroundColor3 = C.PANEL2}, 0.15) end)
    end)

    return row, setOn
end

local function switchTab(name)
    State.ActiveTab = name
    -- Update button styles
    for _, t in ipairs(TABS) do
        local btn = TabButtons[t]
        if btn then
            local isActive = t == name
            tween(btn, {BackgroundColor3 = isActive
                and Color3.fromRGB(15,22,40)
                or  C.PANEL2}, 0.15)
            btn.TextColor3 = isActive and (tabColors[t] or C.WAVE) or C.SUBTEXT
        end
    end
    -- Move tab line
    local idx = table.find(TABS, name) or 1
    local tabW = 1/#TABS
    tween(TabLine, {
        Position = UDim2.new(tabW*(idx-1) + tabW*0.1, 0, 1, -2),
        Size     = UDim2.new(tabW*0.8, 0, 0, 2),
        BackgroundColor3 = tabColors[name] or C.WAVE,
    }, 0.2)

    -- Load page
    if Pages[name] then Pages[name]() end
end

for _, t in ipairs(TABS) do
    TabButtons[t].MouseButton1Click:Connect(function() switchTab(t) end)
end

--  DRAGGING (touch + mouse)
-- â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
do
    local dragging, dragStart, startPos
    local function onInput(input)
        local pos = input.Position
        dragging = true
        dragStart = pos
        startPos = MainFrame.Position
    end
    local function onMove(input)
        if dragging then
            local delta = input.Position - dragStart
            MainFrame.Position = UDim2.new(
                startPos.X.Scale, startPos.X.Offset + delta.X,
                startPos.Y.Scale, startPos.Y.Offset + delta.Y)
        end
    end
    local function onEnd() dragging = false end
    TitleBar.InputBegan:Connect(onInput)
    TitleBar.InputChanged:Connect(onMove)
    TitleBar.InputEnded:Connect(onEnd)
    UserInputService.TouchMoved:Connect(onMove)
    UserInputService.TouchEnded:Connect(onEnd)
end

        if hum.WalkSpeed < 14 then hum.WalkSpeed = 16 end
    end

        for _, p in ipairs(char:GetDescendants()) do
            if p:IsA("BasePart") then
                p.CanCollide = false
            end
        end
    end

    -- FLY MODE
    if State.FlyMode and root then
        local bv = root:FindFirstChild("FlyBV")
        if bv then
            local speed = UserInputService:IsKeyDown(Enum.KeyCode.W) and -30 or 0
            bv.Velocity = Vector3.new(0, 0, speed)
        end
    end

    -- HEALTH LOCK (Layer 1 God Mode)
    if State.HealthLock and hum then
        if hum.MaxHealth ~= math.huge then hum.MaxHealth = math.huge end
        if hum.Health < hum.MaxHealth then
            hum.Health = hum.MaxHealth
        end
    end

    -- INVULN â€” hook TakeDamage each frame in case it resets
    if State.Invuln and hum then
        hum.TakeDamage = function() end
    end

    -- ANTI-GRAVITY (Layer 5)

        if root.Position.Y < 8 then
            root.CFrame = CFrame.new(root.Position.X, 9, root.Position.Z)
        end
    end

    if State.AutoTele and root then
        for _, obj in ipairs(workspace:GetDescendants()) do
            if obj:IsA("BasePart") then
                local n = obj.Name:lower()
                if n:find("tsunami") or n:find("wave") or n:find("water") then
                    local dist = (obj.Position - root.Position).Magnitude
                    if dist < 22 then
                        root.CFrame = root.CFrame * CFrame.new(0, 4, -55)
                        State.KillsBlocked += 1
                        break
                    end
                end
            end

                        bv.MaxForce = Vector3.new(1e9,1e9,1e9)

                local n = obj.Name:lower()
                if n:find("coin") or n:find("gem") or n:find("pickup") or n:find("collect") then
                    local dist = (obj.Position - root.Position).Magnitude
                    if dist < 30 then

                    end
                end

        end
