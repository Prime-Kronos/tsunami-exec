local function applyGod(on)
    pcall(function()
        local c = getChar() if not c then return end
        for _, p in ipairs(c:GetDescendants()) do
            if p:IsA("BasePart") then
                p.LocalTransparencyModifier = on and 1 or 0
                if on then p.CanCollide = false end
            end
        end
        local h = getHum()
        if h and on then h.MaxHealth=1e6 h.Health=1e6 h.BreakJointsOnDeath=false end
    end)
end

local function isCoin(o)
    if not (o:IsA("BasePart") or o:IsA("MeshPart") or o:IsA("UnionOperation")) then return false end
    local n = o.Name:lower()
    return (n=="coin" or n:find("coin") or n=="dropcoin" or n=="goldcoin" or n=="silvercoin")
        and o.Parent ~= nil
        and o.Transparency < 0.9
        and o.Parent ~= LP.Character
end

-- Coin Farm: HRP Ñ‚ĞµĞ»ĞµĞ¿Ğ¾Ñ€Ñ‚Ğ¸Ñ€ÑƒĞµÑ‚ÑÑ Ğº ĞºĞ°Ğ¶Ğ´Ğ¾Ğ¹ Ğ¼Ğ¾Ğ½ĞµÑ‚Ğµ Ğ¿Ğ¾ Ğ¾Ñ‡ĞµÑ€ĞµĞ´Ğ¸
task.spawn(function()
    while task.wait(0.05) do
        if not CFG.coinFarm then continue end
        local hrp = getHRP()
        local hum = getHum()
        if not hrp or not hum then continue end

        local coins = {}
        for _, o in ipairs(workspace:GetDescendants()) do
            if isCoin(o) then table.insert(coins, o) end
        end

        for _, coin in ipairs(coins) do
            if not CFG.coinFarm then break end
            pcall(function()
                if coin and coin.Parent then
                    -- Ğ¢ĞµĞ»ĞµĞ¿Ğ¾Ñ€Ñ‚ Ğ¸Ğ³Ñ€Ğ¾ĞºĞ° ĞŸĞ Ğ¯ĞœĞ ĞĞ Ğ¼Ğ¾Ğ½ĞµÑ‚Ñƒ
                    hrp.CFrame = CFrame.new(coin.Position) * CFrame.new(0, 2, 0)
                    task.wait(0.04)
                    -- Ğ˜Ğ¼Ğ¸Ñ‚Ğ°Ñ†Ğ¸Ñ Touch Ñ‡ĞµÑ€ĞµĞ· FireTouched ĞµÑĞ»Ğ¸ ĞµÑÑ‚ÑŒ ÑĞºÑ€Ğ¸Ğ¿Ñ‚
                    for _, child in ipairs(coin:GetChildren()) do
                        if child:IsA("Script") or child:IsA("LocalScript") then
                            pcall(function()
                                child:FindFirstChildOfClass("BindableEvent") and
                                child:FindFirstChildOfClass("BindableEvent"):Fire(hrp)
                            end)
                        end
                    end
                    -- ĞŸÑ€Ğ¸Ğ½ÑƒĞ´Ğ¸Ñ‚ĞµĞ»ÑŒĞ½Ñ‹Ğ¹ Touch
                    local touchConn
                    touchConn = coin.Touched:Connect(function() end)
                    hrp.CFrame = CFrame.new(coin.Position)
                    task.wait(0.02)
                    pcall(function() touchConn:Disconnect() end)
                end
            end)
            task.wait(0.06)
        end
    end
end)

local ROLE_COLORS={Murderer=Color3.fromRGB(255,60,60),Sheriff=Color3.fromRGB(60,140,255),Innocent=Color3.fromRGB(60,230,110)}
local ROLE_LABELS={Murderer="ğŸ”ª Ğ£Ğ‘Ğ˜Ğ™Ğ¦Ğ",Sheriff="ğŸ”« Ğ¨Ğ•Ğ Ğ˜Ğ¤",Innocent="ğŸ˜‡ ĞĞ•Ğ’Ğ˜ĞĞĞ«Ğ™"}

local function getRole(player)
    local role="Innocent"
    pcall(function()
        local char=player.Character if not char then return end
        for _,item in ipairs(char:GetChildren()) do
            if item:IsA("Tool") then
                local n=item.Name:lower()
                if n:find("knife") or n:find("murder") or n:find("blade") then role="Murderer" return
                elseif n:find("gun") or n:find("sheriff") or n:find("revolver") then role="Sheriff" return end
            end
        end
        local bp=player:FindFirstChild("Backpack")
        if bp then for _,item in ipairs(bp:GetChildren()) do
            if item:IsA("Tool") then
                local n=item.Name:lower()
                if n:find("knife") or n:find("murder") then role="Murderer" return
                elseif n:find("gun") or n:find("sheriff") then role="Sheriff" return end
            end
        end end
    end)
    return role
end

local function removeESP(p)
    if espObjects[p] then pcall(function() espObjects[p]:Destroy() end) espObjects[p]=nil end
end

        local function upd()
            if not bb.Parent then return end
            local role=getRole(p) local col=ROLE_COLORS[role]
            nL.TextColor3=col rL.Text=ROLE_LABELS[role] rL.TextColor3=col
            local hp=math.max(0,math.min(100,math.floor(hum.Health)))
            hL.Text="â¤ HP: "..hp
            local myH=getHRP()
            if myH then dL.Text="ğŸ“ "..math.floor((myH.Position-hrp.Position).Magnitude).." studs" end

    end
    if p.Character then task.spawn(function() setup(p.Character) end) end
    p.CharacterAdded:Connect(function(c) task.wait(1) task.spawn(function() setup(c) end) end)
end

        roleTag.Text=(role=="Murderer" and "ğŸ”ª" or role=="Sheriff" and "ğŸ”«" or "ğŸ˜‡")

    end
end
